Country heatmap

Michael Beer

library(tidyverse)
library("ggplot2")
library("sf")
library("rnaturalearth")
library("rnaturalearthdata")
theme_set(theme_void())
suicides <- read_csv("../data/master.csv")
world <- ne_countries(scale = "medium", returnclass = "sf") %>%
  filter(type == "Sovereign country" | type == "Country")
  suicide_rates <- suicides %>%
  group_by(country) %>% 
  rename(suicides_per_100k_pop = 'suicides/100k pop',
         name_long = 'country') %>%
  summarise(suicide_rates = sum(suicides_per_100k_pop) / 30) %>% 
  arrange(desc(suicide_rates))
## `summarise()` ungrouping output (override with `.groups` argument)
  suicidemap <- merge(world, suicide_rates, by="name_long", all=TRUE)
  ggplot(data = suicidemap) +
    geom_sf(aes(fill = suicide_rates)) +
        labs(title = "Suicide Rates Per Country", fill = "Suicide Rates Per \n 100k People") +
           scale_fill_continuous(low="thistle2", high="darkred", 
                       guide="colorbar",na.value="grey") +
                            theme(panel.background = element_rect(fill = "ivory", color = "black"))
